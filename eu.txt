{
    def comb(n,r){
        var ans = -1;
        if(r==0){
            ans = 1;
        }
        var num = 1;
        for(i=n-r+1;i<n+1;i=i+1){
            num = num * i;
        }
        var den = 1;
        for(i=1;i<r+1;i=i+1){
            den = den * i;
        }

        if (ans == -1){
            ans = num // den;
        }

        return ans;
    }

    print comb(40,20);
}

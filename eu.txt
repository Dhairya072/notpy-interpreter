{
    def factorial(n){
        var ans = -1;
        if(n==1){
            ans = 1;
        }

        if (ans == -1){
            ans = n * factorial(n-1);
        }
        return ans;
    }
    def sumDigits(n){
      var ans = -2;
        if(n<10){
            ans = n;
        }

        if (ans != -2){
            ans = n%10 + sumDigits(n//10);
        }
        return ans;
    }

    var ans2 = sumDigits(factorial(100));
    print ans2;
}

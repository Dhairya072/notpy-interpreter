var s = " one  two three four five six seven eight nine ten  ";
var len = s.length;
var count = 0;
for(i = 1; i < len; i = i + 1)
{
    var prev = i - 1;
    if(s[i] != " " and s[prev] == " ")
    {
        count = count + 1;
    }
}
if(s[0] != " ")
{
    count = count + 1;
}
print count;
